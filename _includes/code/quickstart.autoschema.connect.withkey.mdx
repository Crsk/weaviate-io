```bash
echo '{
  "query": "<QUERY>"

    -X POST \
    -H 'Content-Type: application/json' \
    -H "X-OpenAI-Api-Key: <API-KEY>" \
    -d @- \
    https://some-endpoint.weaviate.network/v1/graphql
```
